def triangle_minimum_path_sum(triangle: list[list[int]]) -> int:

    n, memo = len(triangle), {}

    def dp(r, c):
        if r == n - 1:
            return triangle[r][c]
        if (r, c) not in memo:
            memo[(r, c)] = triangle[r][c] + min(dp(r + 1, c), dp(r + 1, c + 1))
        return memo[(r, c)]

    return dp(0, 0)
