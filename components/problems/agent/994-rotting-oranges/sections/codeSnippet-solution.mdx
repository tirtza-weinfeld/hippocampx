from collections import deque

def rotting_oranges(grid: list[list[int]]) -> int:
    m, n = len(grid), len(grid[0])
    q, fresh = deque(), 0
    dirs = [(0,1),(0,-1),(1,0),(-1,0)]

    for i in range(m):
        for j in range(n):
            if grid[i][j] == 2:
                q.append((i, j))
            elif grid[i][j] == 1:
                fresh += 1
    if fresh == 0:
        return 0

    minutes = 0
    while q and fresh:
        minutes += 1
        for _ in range(len(q)):
            x, y = q.popleft()
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if 0 <= nx < m and 0 <= ny < n and grid[nx][ny] == 1:
                    grid[nx][ny] = 2
                    fresh -= 1
                    q.append((nx, ny))

    return minutes if fresh == 0 else -1
