class Solution:
    def canVisitAllRooms(self, rooms: list[list[int]]) -> bool:
        stack = [0]
        seen = [True] + [False] * (len(rooms) - 1)
        while stack:
            for key in rooms[stack.pop()]:
                if not seen[key]:
                    seen[key] = True
                    stack.append(key)
        return all(seen)
