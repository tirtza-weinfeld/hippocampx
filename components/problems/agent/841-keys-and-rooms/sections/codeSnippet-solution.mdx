class Solution:
    def canVisitAllRooms(self, rooms: list[list[int]]) -> bool:

        seen = [False] * len(rooms); stack = [0]
        seen[0] = True
        while stack:
            for key in rooms[stack.pop()]:
                if not seen[key]:
                    seen[key] = True
                    stack.append(key)
        return all(seen)
