def merge(intervals: list[list[int]]) -> list[list[int]]:
    intervals.sort()
    merged = []
    start, end = intervals[0]
    for s, e in intervals[1:]:
        if s > end:
            merged.append([start, end])
            start, end = s, e
        else:
            end = max(e, end)
    merged.append(start,end)        
    return merged
