```python meta="source=problems/2140-solving-questions-with-brainpower/solution.py"
def most_points_solving_questions_with_brainpower(questions: list[list[int]]) -> int:

    memo, n = {}, len(questions)

    def dp(i):
        if i >= n:
            return 0

        if i not in memo:
            memo[i] = max(dp(i + 1), (q:=questions[i])[0] + dp(i + 1 + q[1]))
        return memo[i]

    return dp(0)
```
